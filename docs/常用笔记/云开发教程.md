# 完整代码

```vue
<template>
  <view class="container">
    <button @click="init(0)">status=0</button>
    <button @click="init(1)">status=1</button>
    <button @click="init(2)">status=2</button>
  </view>
</template>

<script>
  export default {
    methods: {
      init(e) {
        const db = wx.cloud.database();
        db.collection('life-numbers').where({
          status:e
        }).get({
          success: (res => {
            console.log(res);
          })
        })
      }
    }
  }
</script>
```
